\section{karnaugh-map}
        \begin{karnaugh-map}[4][4][1][$ZW$][$XY$]
        \minterms{0,1,4,5,6,7,8,9,11,15}
        \maxterms{2,3,10,12,13,14}
        \implicant{0}{5}
        \implicantedge{0}{1}{8}{9}
        \implicant{4}{6}
        \implicant{15}{11}
        \end{karnaugh-map}

        F=X'Z'+Y'Z'+X'Y+XZW
